var flag = false;
function sortbyname(key) {
    $("#tbody").empty();
    
    if (flag == false) {
        flag = true;
    }
    else {
        flag = false;
    }
    stats.sort(function (a, b) {
       
        if (a[key]> b[key]) {
            return flag ? 1 : -1;
        }
        if (a[key] < b[key]) {
            return flag ? -1 : 1;
        }
        else { return 0;}
    })
    load();
}  
function load() {
    for (i in stats) {
        var tr = $("")
        for (j in stats[i]) {
            var td = $("");
            td.append(stats[i][j]);
            tr.append(td);
        }
        var tb = $("#tbody");
        tb.append(tr);
    }
}
